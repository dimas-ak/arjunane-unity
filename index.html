Get Game Object

Let's say you have some Components:
  • MainCamera
  • Directional Light
  • Canvas
    • Button_go
    • Button_exit
    • Buttons
    • Buttons
    • RawImage
    • Panel_main
    • Panel
      • Text_username
      • InputField_username
  • 3D_object

Get

Get Your game object 
var panel_main = ar.Get("Panel_main");

panel_main.GameObject; // to get Game Object

var buttons = ar.Get("Buttons");

buttons.GameObjects; // to get all Game Objects
// NOTE: if you use buttons.GameObject instead of GameObjects, it will catch first GameObject of (from example) Buttons
Set your GameObject(s) 
public class Example : MonoBehaviour
{
    public GameObject   _gameObject;
    public GameObject[] _gameObjects;

    void Start()
    {
        ar.Get(_gameObject);
        ar.Get(_gameObjects);
    }
}

Find

Find your GameObject(s): 
var panel = ar.Get("Panel");
panel.Find("Text_username");
Set your GameObject(s) to find Your GameObject(s) 
public class Example : MonoBehaviour
{
    public GameObject _parentGameObject;

    void Start()
    {
        var find = ar.Find(_parentGameObject, "Text_password");
    }
}

ForEach

Looping your GameObjects 

var panels = ar.Get("Panel");
panels.ForEach( (go, i) => {
    GameObject gameObject = go;
    int index = i;
});

Set Active

Set visible to your GameObject(s) 

var panels = ar.Get("Panel");
panels.SetActive(false);

Index Of

Get a GameObject for certain index 
int index = 5;
var panels = ar.Get("Panel");
panels.IndexOf(index).GameObject;

Set

Method Set is used to Get Component of the Game Object 
ar.Get("Panels").Animator(); // get component Animator and return it
ar.Get("Username_field").InputField(); // get component Input Field and return it
ar.Get("User_RawImage").RawImage(); // get component Raw Image and return it
ar.Get("User_text").Text(); // get component Text and return it

Color From Hex

Method ColorFromHex is used to set color from Hex (string) 
Color white_color = ar.ColorFromHex("#FFFFFF");

Insert Game Object(s)

Insert or add your prefab(s) 
public class Example : MonoBehaviour
{
    public GameObject prefabGameObject;
    public GameObject[] prefabGameObjects = new GameObject[10];

    void Start()
    {
        var position = new Vector3(0,0,0);
        ar.InsertGameObject(prefabGameObject, position, Quaternion.identity);

        ar.Get(prefabGameObjects).ForEach( (go, i) => {
            ar.InsertGameObject(go, position, Quaternion.identity);
        });
    }
}

Clone

Clone GameObjects
var parent = ar.Get("Parent_panel").GameObject;
var child  = ar.Get("Child_panel").GameObject;

ar.Clone(parent, child, 1);


With ForEach
GameObject parent   = ar.Get("Parent_panel").SetEl();

ar.Get("Childs").ForEach(delegate(GameObject ini, int i) {
    ar.Clone(parent, ini);
});



ar.Clone(
    GameObject parent,
    GameObject child,
    int how_many ( = default is 1),
    bool world_position ( = default is false)
);

Web

NOTE: Please use HTTPS, because sometimes it didn't work if you use HTTP for request or Unity will show Error if you use HTTP for online request
Get string from your Website 
StartCoroutine( ar.GetURL("your-url").sendWebRequest( (result, is_success, error) => {
    if(is_success)
    {
        // your action or convert the 'result' to JSON
    }
    else
    {
        // Error : your handled
    }
}) );
Send input POST 
InputField username = ar.Get("input_username").InputField();
InputField password = ar.Get("input_password").InputField();

var form = ar.GetURL("your_url").AddField("username", username.text)
                        .AddField("password", password.text);


OR

Dictionary<string, string> dr = new Dictionary <string, string>();
dr.Add("username", username.text);
dr.Add("password", password.text);
var form = ar.GetURL("your_url").AddFields(dr);


Send it to your Website

StartCoroutine( form.SendForm( (result, is_success, error) => {
    if(is_success)
    {
        // your action
    }
    else
    {
    // Error : your handled
    }
}) );


SendForm( () => {
    string result,
    bool is_success,
    string error,
    int time_out ( = Default is 60 in second(s) )
});
